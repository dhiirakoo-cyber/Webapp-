Manual Client-Side Configuration

Entering these will persist them in your local browser, allowing live Supabase database queries and writes directly on GitHub Pages with zero server dependencies.

Copy and run this schema script inside your Supabase SQL Editor to instantly bootstrap the required database structure:

-- Amoo Academy Database Schema
CREATE TABLE courses (
  id TEXT PRIMARY KEY,
  title TEXT NOT NULL,
  category TEXT,
  rating NUMERIC,
  price NUMERIC,
  reviews INTEGER,
  duration TEXT,
  level TEXT,
  image TEXT,
  description TEXT,
  curriculum JSONB,
  benefits JSONB
);

CREATE TABLE students (
  email TEXT PRIMARY KEY,
  username TEXT,
  name TEXT,
  phone TEXT,
  password TEXT,
  role TEXT
);

CREATE TABLE enrollments (
  id TEXT PRIMARY KEY,
  "studentName" TEXT,
  "studentEmail" TEXT,
  "studentPhone" TEXT,
  "courseId" TEXT,
  "courseName" TEXT,
  price NUMERIC,
  "paymentMethod" TEXT,
  "paymentDetails" TEXT,
  status TEXT,
  date TEXT
);
